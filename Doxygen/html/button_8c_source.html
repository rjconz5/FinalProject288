55 uint8_t button_getButton() {
56 
57  #warning "Unimplemented function: uint8_t button_getButton(void)" // delete warning after implementing
58 
59  //
60  // DELETE ME - How bitmasking works
61  // ----------------------------------------
62  // In embedded programming, often we only care about one or a few bits in a piece of
63  // data. There are several bitwise operators that we can apply to data in order
64  // to "mask" the bits that we don't care about.
65  //
66  // | = bitwise OR & = bitwise AND ^ = bitwise XOR ~ = bitwise NOT
67  // << x = shift left by x bits >> x = shift right by x bits
68  //
69  // Let's say we want to know if push button 3 (SW3) of GPIO_PORTE_DATA_R is
70  // pushed. Since push buttons are high (1) initially, and low (0) if pushed, PORTE should
71  // look like:
72  // GPIO_PORTE_DATA_R => 0b???? ?0?? if SW3 is pushed
73  // GPIO_PORTE_DATA_R => 0b???? ?1?? if SW3 is not pushed
74  //
75  // This is not useful: There are 128 different 8 bit numbers that have the 3rd bit high or low.
76  // We can make it more clear if we mask the other 7 bits:
77  //
78  // Bitwise AND:
79  // (GPIO_PORTE_DATA_R & 0b0000 0100) == 0b00000000 if SW3 is pushed
80  // (GPIO_PORTE_DATA_R & 0b0000 0100) == 0b00000100 if SW3 is not pushed
81  //
82  // Bitwise OR:
83  // (GPIO_PORTE_DATA_R | 0b1111 1011) == 0b11111011 if SW3 is pushed
84  // (GPIO_PORTE_DATA_R | 0b1111 1011) == 0b11111111 if SW3 is not pushed
85  //
86  // Other techniques (Shifting and bitwise AND)
87  // ((GPIO_PORTE_DATA_R >> 2) & 1) == 0 if SW3 is pushed
88  // ((GPIO_PORTE_DATA_R >> 2) & 0b00000001) == 1 if SW3 is not pushed
89 
90  if(((GPIO_PORTE_DATA_R) & 0b00100000) == 0){
91  return 6;
92  }else if(((GPIO_PORTE_DATA_R) & 0b00010000) == 0){
93  return 5;
94  }else if(((GPIO_PORTE_DATA_R) & 0b00001000) == 0){
95  return 4;
96  }else if(((GPIO_PORTE_DATA_R) & 0b00000100) == 0){
97  return 3;
98  }else if(((GPIO_PORTE_DATA_R) & 0b00000010) == 0){
99  return 2;
100  }else if(((GPIO_PORTE_DATA_R) & 0b00000001) == 0){
101  return 1;
102  }else{
103  return 0;
104  }
105 }
